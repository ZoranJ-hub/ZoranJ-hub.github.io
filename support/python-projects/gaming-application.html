Challenge

Design and implement a responsive game with smooth animations, collision detection, event handling, and extensible architecture that can support multiple game entities and complex interactions.

Solution

Developed a modular game framework using Pygame with object-oriented architecture, implementing a main game loop with fixed timestep, sprite-based collision detection, input handling, and entity management. Designed the system for extensibility, allowing easy addition of new game objects and mechanics.

Technical Deep Dive

  1. Architected game loop with fixed timestep (60 FPS) ensuring consistent physics and smooth gameplay across different hardware.
  2. Implemented sprite-based entity system with inheritance hierarchy: GameObject → Player/Enemy/Projectile, each managing its own rendering and updates.
  3. Developed AABB (Axis-Aligned Bounding Box) collision detection with spatial optimization to handle multiple entities efficiently.
  4. Built event-driven input handler supporting keyboard/mouse input with rebindable controls.
  5. Managed game state machine (Menu → Playing → Paused → GameOver) for transitions and mode-specific logic.
  6. Created particle systems for visual effects (explosions, projectiles, impacts) using pooling for performance optimization.
  7. Integrated score tracking, persistence, and leaderboard system with JSON-based save files.

Python implementation demonstrating game loop, entity management, and collision detection.

import pygame
import json
from enum import Enum
from abc import ABC, abstractmethod
from typing import List, Tuple

# Constants
SCREEN_WIDTH = 800
SCREEN_HEIGHT = 600
FPS = 60
TIMESTEP = 1.0 / FPS

class GameState(Enum):
    MENU = 1
    PLAYING = 2
    PAUSED = 3
    GAME_OVER = 4

class GameObject(ABC):
    """Base class for all game entities."""
    def __init__(self, x: float, y: float, width: int, height: int):
        self.x = x
        self.y = y
        self.width = width
        self.height = height
        self.vx = 0.0  # Velocity X
        self.vy = 0.0  # Velocity Y
        self.active = True
    
    @abstractmethod
    def update(self, dt: float):
        """Update entity state."""
        pass
    
    @abstractmethod
    def draw(self, surface: pygame.Surface):
        """Render entity to surface."""
        pass
    
    def get_rect(self) -> pygame.Rect:
        """Get bounding box for collision detection."""
        return pygame.Rect(self.x, self.y, self.width, self.height)

class Player(GameObject):
    """Player-controlled character."""
    def __init__(self, x: float, y: float):
        super().__init__(x, y, 30, 40)
        self.speed = 250  # pixels/second
        self.health = 100
        self.score = 0
    
    def handle_input(self, keys):
        """Process keyboard input."""
        self.vx = 0
        if keys[pygame.K_LEFT] or keys[pygame.K_a]:
            self.vx = -self.speed
        if keys[pygame.K_RIGHT] or keys[pygame.K_d]:
            self.vx = self.speed
    
    def update(self, dt: float):
        """Update player position with boundary checking."""
        self.x += self.vx * dt
        # Clamp to screen bounds
        self.x = max(0, min(self.x, SCREEN_WIDTH - self.width))
    
    def draw(self, surface: pygame.Surface):
        """Draw player as a rect."""
        pygame.draw.rect(surface, (0, 255, 0), self.get_rect())
        # Draw health bar above player
        health_bar_width = self.width
        health_ratio = max(0, self.health / 100)
        pygame.draw.rect(surface, (255, 0, 0), 
                        (self.x, self.y - 10, health_bar_width, 5))
        pygame.draw.rect(surface, (0, 255, 0), 
                        (self.x, self.y - 10, health_bar_width * health_ratio, 5))

class Enemy(GameObject):
    """Enemy entity with simple AI."""
    def __init__(self, x: float, y: float):
        super().__init__(x, y, 25, 25)
        self.speed = 100
        self.direction = 1
    
    def update(self, dt: float):
        """Simple left-right patrol."""
        self.x += self.speed * self.direction * dt
        # Bounce at screen edges
        if self.x <= 0 or self.x + self.width >= SCREEN_WIDTH:
            self.direction *= -1
    
    def draw(self, surface: pygame.Surface):
        pygame.draw.rect(surface, (255, 0, 0), self.get_rect())

class Projectile(GameObject):
    """Player projectile."""
    def __init__(self, x: float, y: float, vx: float, vy: float):
        super().__init__(x, y, 5, 10)
        self.vx = vx
        self.vy = vy
    
    def update(self, dt: float):
        self.x += self.vx * dt
        self.y += self.vy * dt
        # Deactivate if off-screen
        if (self.x < 0 or self.x > SCREEN_WIDTH or 
            self.y < 0 or self.y > SCREEN_HEIGHT):
            self.active = False
    
    def draw(self, surface: pygame.Surface):
        pygame.draw.rect(surface, (255, 255, 0), self.get_rect())

class CollisionDetector:
    """AABB collision detection."""
    @staticmethod
    def check_collision(rect1: pygame.Rect, rect2: pygame.Rect) -> bool:
        return rect1.colliderect(rect2)
    
    @staticmethod
    def check_collisions(entity: GameObject, entities: List[GameObject]) -> List[GameObject]:
        """Return list of entities colliding with given entity."""
        collisions = []
        entity_rect = entity.get_rect()
        for other in entities:
            if other is not entity and other.active:
                if CollisionDetector.check_collision(entity_rect, other.get_rect()):
                    collisions.append(other)
        return collisions

class Game:
    """Main game engine."""
    def __init__(self):
        pygame.init()
        self.screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
        pygame.display.set_caption("Python Game Engine")
        self.clock = pygame.time.Clock()
        self.running = True
        self.state = GameState.PLAYING
        
        # Game entities
        self.player = Player(SCREEN_WIDTH // 2, SCREEN_HEIGHT - 50)
        self.enemies: List[Enemy] = [Enemy(100, 100), Enemy(400, 150), Enemy(600, 120)]
        self.projectiles: List[Projectile] = []
        self.all_entities: List[GameObject] = [self.player] + self.enemies + self.projectiles
        
        # Game stats
        self.score = 0
        self.wave = 1
    
    def handle_events(self):
        """Process input and window events."""
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                self.running = False
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_ESCAPE:
                    self.state = GameState.PAUSED if self.state == GameState.PLAYING else GameState.PLAYING
                elif event.key == pygame.K_SPACE and self.state == GameState.PLAYING:
                    # Fire projectile
                    projectile = Projectile(
                        self.player.x + self.player.width // 2,
                        self.player.y,
                        vx=0,
                        vy=-400
                    )
                    self.projectiles.append(projectile)
                    self.all_entities.append(projectile)
    
    def update(self, dt: float):
        """Update game state."""
        if self.state != GameState.PLAYING:
            return
        
        keys = pygame.key.get_pressed()
        self.player.handle_input(keys)
        
        # Update all entities
        for entity in self.all_entities:
            if entity.active:
                entity.update(dt)
        
        # Remove inactive entities
        self.projectiles = [p for p in self.projectiles if p.active]
        self.all_entities = [e for e in self.all_entities if e.active]
        
        # Check projectile-enemy collisions
        for projectile in self.projectiles[:]:
            hit_enemies = CollisionDetector.check_collisions(projectile, self.enemies)
            for enemy in hit_enemies:
                enemy.active = False
                projectile.active = False
                self.score += 10
        
        # Check player-enemy collisions
        player_hits = CollisionDetector.check_collisions(self.player, self.enemies)
        for enemy in player_hits:
            self.player.health -= 10
            if self.player.health <= 0:
                self.state = GameState.GAME_OVER
    
    def render(self):
        """Draw game to screen."""
        self.screen.fill((0, 0, 0))  # Black background
        
        for entity in self.all_entities:
            if entity.active:
                entity.draw(self.screen)
        
        # Draw UI
        font = pygame.font.Font(None, 36)
        score_text = font.render(f"Score: {self.score}", True, (255, 255, 255))
        health_text = font.render(f"Health: {int(self.player.health)}", True, (255, 255, 255))
        self.screen.blit(score_text, (10, 10))
        self.screen.blit(health_text, (10, 50))
        
        if self.state == GameState.PAUSED:
            pause_text = font.render("PAUSED (Press ESC to resume)", True, (255, 255, 0))
            self.screen.blit(pause_text, (SCREEN_WIDTH // 2 - 200, SCREEN_HEIGHT // 2))
        
        pygame.display.flip()
    
    def run(self):
        """Main game loop."""
        while self.running:
            dt = self.clock.tick(FPS) / 1000.0  # Convert ms to seconds
            
            self.handle_events()
            self.update(dt)
            self.render()
        
        pygame.quit()

# Main execution
if __name__ == "__main__":
    game = Game()
    game.run()

class SpatialGrid:
    """Optimize collision detection using spatial partitioning."""
    def __init__(self, cell_size: int, width: int, height: int):
        self.cell_size = cell_size
        self.cols = width // cell_size
        self.rows = height // cell_size
        self.grid = [[[] for _ in range(self.cols)] for _ in range(self.rows)]
    
    def insert(self, entity: GameObject):
        """Insert entity into grid cells."""
        rect = entity.get_rect()
        col_min = int(rect.left // self.cell_size)
        col_max = int(rect.right // self.cell_size)
        row_min = int(rect.top // self.cell_size)
        row_max = int(rect.bottom // self.cell_size)
        
        for row in range(max(0, row_min), min(self.rows, row_max + 1)):
            for col in range(max(0, col_min), min(self.cols, col_max + 1)):
                self.grid[row][col].append(entity)
    
    def query(self, rect: pygame.Rect) -> List[GameObject]:
        """Get potentially colliding entities in region."""
        col_min = int(rect.left // self.cell_size)
        col_max = int(rect.right // self.cell_size)
        row_min = int(rect.top // self.cell_size)
        row_max = int(rect.bottom // self.cell_size)
        
        candidates = set()
        for row in range(max(0, row_min), min(self.rows, row_max + 1)):
            for col in range(max(0, col_min), min(self.cols, col_max + 1)):
                candidates.update(self.grid[row][col])
        
        return list(candidates)
