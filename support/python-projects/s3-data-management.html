Secure AWS S3 Data Management System

Built a Python framework for secure, compliant S3 bucket management with encryption, versioning, and lifecycle policies enforced through IAM and AWS best practices.

Challenge

Manage sensitive data across multiple S3 buckets while maintaining compliance requirements, enforcing encryption, implementing version control, and preventing accidental or malicious data exposure.

Solution

Developed a comprehensive Python system using Boto3 to automate S3 bucket provisioning with enforced encryption (SSE-S3 or KMS), versioning, bucket policies, access logging, and lifecycle rules. Combined with IAM role-based access control and CloudWatch monitoring for security and compliance.

Technical Deep Dive

  1. Provisioned S3 buckets with default encryption (AWS KMS for sensitive data, SSE-S3 for standard data).
  2. Configured versioning on all buckets to maintain audit trails and enable rollback capabilities.
  3. Implemented bucket policies using least-privilege IAM principles, blocking unencrypted uploads via `s3:x-amz-server-side-encryption` condition.
  4. Enabled access logging to a centralized audit bucket with automatic expiration to manage storage costs.
  5. Created lifecycle rules to transition old objects to Glacier/Deep Archive and automatically delete expired data.
  6. Integrated CloudWatch monitoring with SNS notifications for unauthorized access attempts and policy violations.
  7. Built automated tagging and cost allocation tags to track data ownership and departmental usage.

Tools & Technologies

Python • Boto3 • AWS S3 • AWS KMS • AWS IAM • CloudWatch • SNS • Access Logs • Lifecycle Policies • Versioning

Code Snippet Example

Python script demonstrating secure S3 bucket creation and policy enforcement.

import boto3
import json
from botocore.exceptions import ClientError

s3_client = boto3.client('s3')
iam_client = boto3.client('iam')

def create_secure_s3_bucket(bucket_name, environment='prod'):
    """
    Create an S3 bucket with encryption, versioning, and access logging enabled.
    """
    try:
        # Create bucket
        s3_client.create_bucket(
            Bucket=bucket_name,
            CreateBucketConfiguration={'LocationConstraint': 'us-west-2'}
        )
        print(f"✓ Bucket {bucket_name} created")
        
        # Enable versioning
        s3_client.put_bucket_versioning(
            Bucket=bucket_name,
            VersioningConfiguration={'Status': 'Enabled'}
        )
        print(f"✓ Versioning enabled")
        
        # Enable default encryption (KMS for prod, SSE-S3 for dev)
        if environment == 'prod':
            encryption_config = {
                'Rules': [{
                    'ApplyServerSideEncryptionByDefault': {
                        'SSEAlgorithm': 'aws:kms',
                        'KMSMasterKeyID': 'arn:aws:kms:us-west-2:ACCOUNT_ID:key/KEY_ID'
                    },
                    'BucketKeyEnabled': True
                }]
            }
        else:
            encryption_config = {
                'Rules': [{
                    'ApplyServerSideEncryptionByDefault': {'SSEAlgorithm': 'AES256'}
                }]
            }
        
        s3_client.put_bucket_encryption(
            Bucket=bucket_name,
            ServerSideEncryptionConfiguration=encryption_config
        )
        print(f"✓ Encryption configured ({environment} mode)")
        
        # Block public access
        s3_client.put_public_access_block(
            Bucket=bucket_name,
            PublicAccessBlockConfiguration={
                'BlockPublicAcls': True,
                'IgnorePublicAcls': True,
                'BlockPublicPolicy': True,
                'RestrictPublicBuckets': True
            }
        )
        print(f"✓ Public access blocked")
        
        return True
    except ClientError as e:
        print(f"✗ Error creating bucket: {e}")
        return False

def enforce_encryption_policy(bucket_name):
    """
    Apply a bucket policy that requires encrypted uploads.
    """
    bucket_policy = {
        "Version": "2012-10-17",
        "Statement": [
            {
                "Sid": "DenyUnencryptedObjectUploads",
                "Effect": "Deny",
                "Principal": "*",
                "Action": "s3:PutObject",
                "Resource": f"arn:aws:s3:::{bucket_name}/*",
                "Condition": {
                    "StringNotEquals": {
                        "s3:x-amz-server-side-encryption": "aws:kms"
                    }
                }
            },
            {
                "Sid": "DenyIncorrectKMSKey",
                "Effect": "Deny",
                "Principal": "*",
                "Action": "s3:PutObject",
                "Resource": f"arn:aws:s3:::{bucket_name}/*",
                "Condition": {
                    "StringNotEquals": {
                        "s3:x-amz-server-side-encryption-aws-kms-key-id": "arn:aws:kms:us-west-2:ACCOUNT_ID:key/KEY_ID"
                    }
                }
            }
        ]
    }
    
    try:
        s3_client.put_bucket_policy(
            Bucket=bucket_name,
            Policy=json.dumps(bucket_policy)
        )
        print(f"✓ Encryption policy enforced on {bucket_name}")
        return True
    except ClientError as e:
        print(f"✗ Error enforcing policy: {e}")
        return False

def setup_access_logging(bucket_name, log_bucket_name):
    """
    Enable access logging to track all requests to the bucket.
    """
    try:
        s3_client.put_bucket_logging(
            Bucket=bucket_name,
            BucketLoggingStatus={
                'LoggingEnabled': {
                    'TargetBucket': log_bucket_name,
                    'TargetPrefix': f'logs/{bucket_name}/'
                }
            }
        )
        print(f"✓ Access logging enabled (logs → {log_bucket_name})")
        return True
    except ClientError as e:
        print(f"✗ Error enabling access logging: {e}")
        return False

def setup_lifecycle_policy(bucket_name):
    """
    Configure lifecycle rules to manage object retention and archival.
    """
    lifecycle_config = {
        'Rules': [
            {
                'Id': 'Archive-old-objects',
                'Status': 'Enabled',
                'Transitions': [
                    {
                        'Days': 30,
                        'StorageClass': 'STANDARD_IA'
                    },
                    {
                        'Days': 90,
                        'StorageClass': 'GLACIER'
                    }
                ],
                'Expiration': {'Days': 365}
            }
        ]
    }
    
    try:
        s3_client.put_bucket_lifecycle_configuration(
            Bucket=bucket_name,
            LifecycleConfiguration=lifecycle_config
        )
        print(f"✓ Lifecycle policy applied (30d→IA, 90d→Glacier, 365d→Delete)")
        return True
    except ClientError as e:
        print(f"✗ Error applying lifecycle policy: {e}")
        return False

# Main execution
if __name__ == "__main__":
    bucket_name = "my-secure-data-bucket-prod"
    log_bucket_name = "my-audit-logs-bucket-prod"
    
    print(f"\n--- Setting up secure S3 infrastructure for {bucket_name} ---\n")
    
    create_secure_s3_bucket(bucket_name, environment='prod')
    enforce_encryption_policy(bucket_name)
    setup_access_logging(bucket_name, log_bucket_name)
    setup_lifecycle_policy(bucket_name)
    
    print(f"\n✓ S3 bucket {bucket_name} fully configured with security best practices\n")
